import { useState } from "react";
import dynamic from "next/dynamic";

export const DynamicEditor = dynamic(() => import("./components/CodeEditor"), {
  ssr: false,
});

export function CodePractice() {
  const [code, setCode] = useState('// Write your code here');
  const [output, setOutput] = useState('');

const evaluateCode = (codeToEval) => {

    try {

      const logs = [];
      const customConsole = {
        log: (...args) => {
          logs.push(args.map(arg =>
            typeof arg === 'object' ? JSON.stringify(arg) : arg
          ).join(' '));
        }
      };
      const transformedCode = codeToEval.replace(
        /console\.log\(/g,
        'customConsole.log('
      );
      new Function('customConsole', transformedCode)(customConsole);
      return logs.join('\n');
    } catch (error) {
      return `Error: ${error.message}`;
    }

};

const handleRunCode = () => {
const result = evaluateCode(code);
setOutput(result);
};

return (

  <div>
    <DynamicEditor
      value={code}
      onChange={(value) => setCode(value)}
      />

      <button
        onClick={handleRunCode}
        style={{
          marginTop: '15px',
          padding: '6px 12px',
          backgroundColor: '#7f199c',
          color: 'white',
          border: 'none',
          borderRadius: '4px',
          cursor: 'pointer'
        }}
      >
        Run Code
      </button>

      {output && <div style={{ marginTop: '20px' }}>
        <h3>Output:</h3>
        <pre style={{
          backgroundColor: '#f5f5f5',
          padding: '10px',
          borderRadius: '4px',
          fontFamily: 'monospace'
        }}>
          {output}
        </pre>
      </div>}
    </div>

);
}
