import { useState } from "react";
import dynamic from "next/dynamic";

export const DynamicEditor = dynamic(() => import("./components/CodeEditor"), {
  ssr: false,
});

export function CodePractice() {
  const [code, setCode] = useState('// Write your code here');
  const [output, setOutput] = useState('');

const evaluateCode = async (codeToEval) => {

const logs = [];

const customConsole = {

    log: (...args) => {

    logs.push(args.map(arg =>
    typeof arg === 'object' ? JSON.stringify(arg) : String(arg)
    ).join(' '));
    },

    error: (...args) => {
    logs.push(`Error: ${args.map(arg =>
              typeof arg === 'object' ? JSON.stringify(arg) : String(arg)
            ).join(' ')}`);
    }

};

    try {
      const transformedCode = `
        return (async () => {
          ${codeToEval.replace(/console\.(log|error)\(/g, 'customConsole.$1(')}
        })();
      `;

      await new Function('customConsole', transformedCode)(customConsole);
      await new Promise(resolve => setTimeout(resolve, 100));
      return logs.join('\n');
    } catch (error) {
      return `Error: ${error.message}`;
    }

};

const handleRunCode = async () => {
try {

    const result = await evaluateCode(code);
    setOutput(result);

} catch (error) {

    setOutput(`Error: ${error.message}`);

}
};

return (

  <div>
    <DynamicEditor
      value={code}
      onChange={(value) => setCode(value)}
      />

      <button
        onClick={handleRunCode}
        style={{
          marginTop: '15px',
          padding: '6px 12px',
          backgroundColor: '#7f199c',
          color: 'white',
          border: 'none',
          borderRadius: '4px',
          cursor: 'pointer'
        }}
      >
        Run Code
      </button>

      {output && <div style={{ marginTop: '20px' }}>
        <h3>Output:</h3>
        <pre style={{
          backgroundColor: '#f5f5f5',
          padding: '10px',
          borderRadius: '4px',
          fontFamily: 'monospace'
        }}>
          {output}
        </pre>
      </div>}
    </div>

);
}
